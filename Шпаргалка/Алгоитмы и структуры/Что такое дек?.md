```
from collections import deque
def max_sliding_window(nums, k):
    d = deque()
    res = []
    for i, num in enumerate(nums):
        while d and nums[d[-1]] < num:
            d.pop()
        d.append(i)
        if d[0] == i - k:
            d.popleft()
        if i >= k - 1:
            res.append(nums[d[0]])
    return res

```
