FROM - задаёт базовый образ  
RUN - запуск команды во время сборки образа  
CMD - команда при запуске контейнера  
ENTRYPOINT - команда при запуске контейнера  
COPY - копирует файлы и директории из хоста в контейнер  
ENV - устанавливает переменные окружения  
EXPOSE - определение порта контейнера  
VOLUME - создать том для хранения данных контейнера на хосте  
WORKDIR - устанавливает рабочий каталог для последующий инструкций RUN, CMD, COPY - копирование файлов с хоста в контейнер  
ADD - COPY для работы с удаленными url адресами  
USER - устанавливает пользователя, под которым будут выполняться команды внутри контейнера  
LABEL - метаданные к образу
